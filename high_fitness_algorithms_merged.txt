import numpy as np

class AdaptiveDE:
    def __init__(self, budget=10000, dim=10, pop_factor=5, p=0.1):
        self.budget = budget
        self.dim = dim
        self.pop_size = int(pop_factor * dim)
        self.F = 0.5
        self.Cr = 0.9
        self.pop = None
        self.fitness = None
        self.f_opt = np.Inf
        self.x_opt = None
        self.eval_count = 0
        self.p = p  # Probability for mirrored sampling
        self.successful_F = []
        self.successful_Cr = []

    def initialize_population(self, func):
        self.pop = np.random.uniform(func.bounds.lb, func.bounds.ub, size=(self.pop_size, self.dim))
        self.fitness = np.array([func(x) for x in self.pop])
        self.eval_count += self.pop_size
        best_index = np.argmin(self.fitness)
        self.f_opt = self.fitness[best_index]
        self.x_opt = self.pop[best_index]

    def evolve(self, func):
        for i in range(self.pop_size):
            if self.eval_count >= self.budget:
                break

            # Mutation
            idxs = np.random.choice(self.pop_size, 3, replace=False)
            x_r1, x_r2, x_r3 = self.pop[idxs]
            x_mutated = x_r1 + self.F * (x_r2 - x_r3)

            # Jittering to avoid stagnation
            if np.random.rand() < 0.05:
                x_mutated += np.random.normal(0, 0.01, size=self.dim)
            
            # Mirrored sampling
            if np.random.rand() < self.p:
                x_mutated = np.where(x_mutated < func.bounds.lb, func.bounds.lb + (func.bounds.lb - x_mutated), x_mutated)
                x_mutated = np.where(x_mutated > func.bounds.ub, func.bounds.ub - (x_mutated - func.bounds.ub), x_mutated)
            
            x_mutated = np.clip(x_mutated, func.bounds.lb, func.bounds.ub)

            # Crossover
            x_trial = self.pop[i].copy()
            j_rand = np.random.randint(self.dim)
            for j in range(self.dim):
                if np.random.rand() < self.Cr or j == j_rand:
                    x_trial[j] = x_mutated[j]

            # Selection
            f_trial = func(x_trial)
            self.eval_count += 1

            if f_trial < self.fitness[i]:
                self.successful_F.append(self.F)
                self.successful_Cr.append(self.Cr)
                self.fitness[i] = f_trial
                self.pop[i] = x_trial

                if f_trial < self.f_opt:
                    self.f_opt = f_trial
                    self.x_opt = x_trial

        # Adapt F and Cr based on successful values
        if self.successful_F:
            self.F = np.mean(self.successful_F)
            self.Cr = np.mean(self.successful_Cr)
            self.successful_F = []
            self.successful_Cr = []
        else:
            self.F = 0.5  # Reset if no successful updates
            self.Cr = 0.9


    def __call__(self, func):
        self.initialize_population(func)

        while self.eval_count < self.budget:
            self.evolve(func)

        return self.f_opt, self.x_opt

        self.f_opt = np.min(fitness)
        self.x_opt = population[np.argmin(fitness)]
